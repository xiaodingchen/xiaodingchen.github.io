)

type myMap struct {
	rw   sync.RWMutex
	m    sync.Mutex
	data map[string]int
}

func (m *myMap) add(key string, val int) {
	m.rw.Lock()
	log.Println("add", key, val)
	//time.Sleep(time.Second * 3)
	defer m.rw.Unlock()
	m.data[key] = val
}

func (m *myMap) mFind(key string) int {
	m.m.Lock()
	log.Println("mfind", key)
	//time.Sleep(time.Second)
	defer m.m.Unlock()
	return m.data[key]
}

func (m *myMap) rwFind(key string) int {
	m.rw.RLock()
	log.Println("rwfind", key)
	//time.Sleep(time.Second)
	defer m.rw.RUnlock()
	return m.data[key]
}

	wg := sync.WaitGroup{}
	m := &myMap{data: make(map[string]int)}
	key := "a"

			defer wg.Done()

			defer wg.Done()

		}(i)

		go func(i int) {
			defer wg.Done()
			log.Println("rwfor", i, m.rwFind(key))
		}(i)
	}

	wg.Wait()
}
